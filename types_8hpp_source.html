233 std::size_t edge(std::size_t i, std::size_t j) const
234 {
235 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
236 // FEATs structured mesh counts fences, rather than fence posts.
237 // Hence the +- 1 everywhere
238
239 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
240
241 // auxiliary index
242 std::size_t pos = 0;
243
244 // z-direction
245 if(
246 i >= num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) +
247 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
248 {
249 pos = i - num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) -
250 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
251 return (pos / num_slices[2]) + ((pos % num_slices[2] + j) * (num_slices[0] + 1) * (num_slices[1] + 1));
252 }
253 // y-direction
254 else if(i >= (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
255 {
256 pos = i - (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
257 std::size_t x = (pos / num_slices[1]) % (num_slices[0] + 1);
258 std::size_t y = pos % num_slices[1];
259 std::size_t z = (pos / num_slices[1]) / (num_slices[0] + 1);
260 return (z * (num_slices[0] + 1) * (num_slices[1] + 1)) + ((y + j) * (num_slices[0] + 1)) + x;
261 }
262 // x-direction
263 else
264 {
265 return i + j + (i / num_slices[0]);
266 }
267 }

277 std::size_t face(std::size_t i, std::size_t j) const
278 {
279 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
280 // FEATs structured mesh counts fences, rather than fence posts.
281 // Hence the +- 1 everywhere
282
283 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
284
285 std::size_t pos = 0;
286 if(i < num_slices[0] * num_slices[1] * (num_slices[2] + 1))
287 {
288 return i + (i / num_slices[0]) + ((i / (num_slices[0] * num_slices[1])) * (num_slices[0] + 1)) + (j % 2) +
289 ((j / 2) * (num_slices[0] + 1));
290 }
291 else if(
292 i >= num_slices[0] * num_slices[1] * (num_slices[2] + 1) &&
293 i < num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1)))
294 {
295 pos = i - num_slices[0] * num_slices[1] * (num_slices[2] + 1);
296 return (pos % num_slices[0]) + ((pos / (num_slices[0] * num_slices[2])) * (num_slices[0] + 1)) +
297 ((pos % (num_slices[0] * num_slices[2])) / num_slices[0] * (num_slices[0] + 1) * (num_slices[1] + 1)) +
298 (j % 2) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
299 }
300 else
301 {
302 pos = i - num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1));
303 return (pos / (num_slices[1] * num_slices[2])) + ((pos % num_slices[1]) * (num_slices[0] + 1)) +
304 (((pos / num_slices[1]) % num_slices[2]) * (num_slices[0] + 1) * (num_slices[1] + 1)) +
305 ((j % 2) * (num_slices[0] + 1)) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
306 }
307 }

317 std::size_t cell(std::size_t idx, std::size_t vert) const
318 {
319 const std::size_t z_layer = idx / ((_xs.size() - 1) * (_ys.size() - 1));
320 const std::size_t y_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) / (_xs.size() - 1);
321 const std::size_t x_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) % (_xs.size() - 1);
322
323 const std::size_t base = (z_layer * (_xs.size() * _ys.size())) + (y_layer * _xs.size()) + x_layer;
324 const std::size_t z_offset = _xs.size() * _ys.size();
325
326 switch(vert)
327 {
328 case 0:
329 return base;
330 case 1:
331 return base + 1;
332 case 2:
333 return base + _xs.size();
334 case 3:
335 return base + _xs.size() + 1;
336 case 4:
337 return base + z_offset;
338 case 5:
339 return base + 1 + z_offset;
340 case 6:
341 return base + _xs.size() + z_offset;
342 case 7:
343 return base + _xs.size() + 1 + z_offset;
344 default:
345 std::abort();
346 }
347 }

356 std::uint64_t subdivision_level(std::size_t idx)
357 {
358 Point vtx = vertex(idx);
359
360 std::uint64_t level = 0;
361
362 for(std::size_t slice(0); slice < _zs.size() - 1; slice++)
363 {
364 if(_zs[slice].coord <= vtx.z && vtx.z <= _zs[slice + 1].coord)
365 {
366 level = std::max(_zs[slice].subdivision_level, level);
367 }
368 }
369
370 return level;
371 }

381 void write_feat_xml(std::ostream& stream) const
382 {
383 stream << "<FeatMeshFile version=\"1\" mesh=\"conformal:hypercube:3:3\">\n";
384 stream << "<Mesh type=\"conformal:hypercube:3:3\" size=\""
385 << num_vertices() << " "
386 << num_edges() << " "
387 << num_faces() << " "
388 << num_cells() << "\">\n";
389 stream << "<Vertices>\n";
390 for(std::size_t i(0); i < num_vertices(); i++)
391 {
392 const Point point = vertex(i);
393 stream << point.x << " " << point.y << " " << point.z << "\n";
394 }
395 stream << "</Vertices>\n";
396
397 stream << "<Topology dim=\"1\">\n";
398 for(std::size_t i(0); i < num_edges(); i++)
399 {
400 for(std::size_t j(0); j < 2; j++)
401 {
402 stream << edge(i, j);
403 stream << (j == 1 ? "\n" : " ");
404 }
405 }
406 stream << "</Topology>\n";
407
408 stream << "<Topology dim=\"2\">\n";
409 for(std::size_t i(0); i < num_faces(); i++)
410 {
411 for(std::size_t j(0); j < 4; j++)
412 {
413 stream << face(i, j);
414 stream << (j == 3 ? "\n" : " ");
415 }
416 }
417 stream << "</Topology>\n";
418
419 stream << "<Topology dim=\"3\">\n";
420 for(std::size_t i(0); i < num_cells(); i++)
421 {
422 for(std::size_t j(0); j < 8; j++)
423 {
424 stream << cell(i, j);
425 stream << (j == 7 ? "\n" : " ");
426 }
427 }
428 stream << "</Topology>\n";
429
430 stream << "</Mesh>\n";
431 stream << "</FeatMeshFile>\n";
432 }
