256 std::size_t edge(std::size_t i, std::size_t j) const
257 {
258 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
259 // FEATs structured mesh counts fences, rather than fence posts.
260 // Hence the +- 1 everywhere
261
262 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
263
264 // auxiliary index
265 std::size_t pos = 0;
266
267 // z-direction
268 if(
269 i >= num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) +
270 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
271 {
272 pos = i - num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) -
273 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
274 return (pos / num_slices[2]) + ((pos % num_slices[2] + j) * (num_slices[0] + 1) * (num_slices[1] + 1));
275 }
276 // y-direction
277 else if(i >= (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
278 {
279 pos = i - (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
280 std::size_t x = (pos / num_slices[1]) % (num_slices[0] + 1);
281 std::size_t y = pos % num_slices[1];
282 std::size_t z = (pos / num_slices[1]) / (num_slices[0] + 1);
283 return (z * (num_slices[0] + 1) * (num_slices[1] + 1)) + ((y + j) * (num_slices[0] + 1)) + x;
284 }
285 // x-direction
286 else
287 {
288 return i + j + (i / num_slices[0]);
289 }
290 }

300 std::size_t face(std::size_t i, std::size_t j) const
301 {
302 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
303 // FEATs structured mesh counts fences, rather than fence posts.
304 // Hence the +- 1 everywhere
305
306 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
307
308 std::size_t pos = 0;
309 if(i < num_slices[0] * num_slices[1] * (num_slices[2] + 1))
310 {
311 return i + (i / num_slices[0]) + ((i / (num_slices[0] * num_slices[1])) * (num_slices[0] + 1)) + (j % 2) +
312 ((j / 2) * (num_slices[0] + 1));
313 }
314 else if(
315 i >= num_slices[0] * num_slices[1] * (num_slices[2] + 1) &&
316 i < num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1)))
317 {
318 pos = i - num_slices[0] * num_slices[1] * (num_slices[2] + 1);
319 return (pos % num_slices[0]) + ((pos / (num_slices[0] * num_slices[2])) * (num_slices[0] + 1)) +
320 ((pos % (num_slices[0] * num_slices[2])) / num_slices[0] * (num_slices[0] + 1) * (num_slices[1] + 1)) +
321 (j % 2) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
322 }
323 else
324 {
325 pos = i - num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1));
326 return (pos / (num_slices[1] * num_slices[2])) + ((pos % num_slices[1]) * (num_slices[0] + 1)) +
327 (((pos / num_slices[1]) % num_slices[2]) * (num_slices[0] + 1) * (num_slices[1] + 1)) +
328 ((j % 2) * (num_slices[0] + 1)) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
329 }
330 }

340 std::size_t cell(std::size_t idx, std::size_t vert) const
341 {
342 const std::size_t z_layer = idx / ((_xs.size() - 1) * (_ys.size() - 1));
343 const std::size_t y_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) / (_xs.size() - 1);
344 const std::size_t x_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) % (_xs.size() - 1);
345
346 const std::size_t base = (z_layer * (_xs.size() * _ys.size())) + (y_layer * _xs.size()) + x_layer;
347 const std::size_t z_offset = _xs.size() * _ys.size();
348
349 switch(vert)
350 {
351 case 0:
352 return base;
353 case 1:
354 return base + 1;
355 case 2:
356 return base + _xs.size();
357 case 3:
358 return base + _xs.size() + 1;
359 case 4:
360 return base + z_offset;
361 case 5:
362 return base + 1 + z_offset;
363 case 6:
364 return base + _xs.size() + z_offset;
365 case 7:
366 return base + _xs.size() + 1 + z_offset;
367 default:
368 std::abort();
369 }
370 }

404 void write_feat_xml(std::ostream& stream) const
405 {
406 stream << "<FeatMeshFile version=\"1\" mesh=\"conformal:hypercube:3:3\">\n";
407 stream << "<Mesh type=\"conformal:hypercube:3:3\" size=\""
408 << num_vertices() << " "
409 << num_edges() << " "
410 << num_faces() << " "
411 << num_cells() << "\">\n";
412 stream << "<Vertices>\n";
413 for(std::size_t i(0); i < num_vertices(); i++)
414 {
415 const Point point = vertex(i);
416 stream << point.x << " " << point.y << " " << point.z << "\n";
417 }
418 stream << "</Vertices>\n";
419
420 stream << "<Topology dim=\"1\">\n";
421 for(std::size_t i(0); i < num_edges(); i++)
422 {
423 for(std::size_t j(0); j < 2; j++)
424 {
425 stream << edge(i, j);
426 stream << (j == 1 ? "\n" : " ");
427 }
428 }
429 stream << "</Topology>\n";
430
431 stream << "<Topology dim=\"2\">\n";
432 for(std::size_t i(0); i < num_faces(); i++)
433 {
434 for(std::size_t j(0); j < 4; j++)
435 {
436 stream << face(i, j);
437 stream << (j == 3 ? "\n" : " ");
438 }
439 }
440 stream << "</Topology>\n";
441
442 stream << "<Topology dim=\"3\">\n";
443 for(std::size_t i(0); i < num_cells(); i++)
444 {
445 for(std::size_t j(0); j < 8; j++)
446 {
447 stream << cell(i, j);
448 stream << (j == 7 ? "\n" : " ");
449 }
450 }
451 stream << "</Topology>\n";
452
453 stream << "</Mesh>\n";
454 stream << "</FeatMeshFile>\n";
455 }
