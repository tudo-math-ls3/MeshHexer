231 std::size_t edge(std::size_t i, std::size_t j) const
232 {
233 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
234 // FEATs structured mesh counts fences, rather than fence posts.
235 // Hence the +- 1 everywhere
236
237 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
238
239 // auxiliary index
240 std::size_t pos = 0;
241
242 // z-direction
243 if(
244 i >= num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) +
245 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
246 {
247 pos = i - num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) -
248 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
249 return (pos / num_slices[2]) + ((pos % num_slices[2] + j) * (num_slices[0] + 1) * (num_slices[1] + 1));
250 }
251 // y-direction
252 else if(i >= (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
253 {
254 pos = i - (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
255 std::size_t x = (pos / num_slices[1]) % (num_slices[0] + 1);
256 std::size_t y = pos % num_slices[1];
257 std::size_t z = (pos / num_slices[1]) / (num_slices[0] + 1);
258 return (z * (num_slices[0] + 1) * (num_slices[1] + 1)) + ((y + j) * (num_slices[0] + 1)) + x;
259 }
260 // x-direction
261 else
262 {
263 return i + j + (i / num_slices[0]);
264 }
265 }

275 std::size_t face(std::size_t i, std::size_t j) const
276 {
277 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
278 // FEATs structured mesh counts fences, rather than fence posts.
279 // Hence the +- 1 everywhere
280
281 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
282
283 std::size_t pos = 0;
284 if(i < num_slices[0] * num_slices[1] * (num_slices[2] + 1))
285 {
286 return i + (i / num_slices[0]) + ((i / (num_slices[0] * num_slices[1])) * (num_slices[0] + 1)) + (j % 2) +
287 ((j / 2) * (num_slices[0] + 1));
288 }
289 else if(
290 i >= num_slices[0] * num_slices[1] * (num_slices[2] + 1) &&
291 i < num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1)))
292 {
293 pos = i - num_slices[0] * num_slices[1] * (num_slices[2] + 1);
294 return (pos % num_slices[0]) + ((pos / (num_slices[0] * num_slices[2])) * (num_slices[0] + 1)) +
295 ((pos % (num_slices[0] * num_slices[2])) / num_slices[0] * (num_slices[0] + 1) * (num_slices[1] + 1)) +
296 (j % 2) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
297 }
298 else
299 {
300 pos = i - num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1));
301 return (pos / (num_slices[1] * num_slices[2])) + ((pos % num_slices[1]) * (num_slices[0] + 1)) +
302 (((pos / num_slices[1]) % num_slices[2]) * (num_slices[0] + 1) * (num_slices[1] + 1)) +
303 ((j % 2) * (num_slices[0] + 1)) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
304 }
305 }

315 std::size_t cell(std::size_t idx, std::size_t vert) const
316 {
317 const std::size_t z_layer = idx / ((_xs.size() - 1) * (_ys.size() - 1));
318 const std::size_t y_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) / (_xs.size() - 1);
319 const std::size_t x_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) % (_xs.size() - 1);
320
321 const std::size_t base = (z_layer * (_xs.size() * _ys.size())) + (y_layer * _xs.size()) + x_layer;
322 const std::size_t z_offset = _xs.size() * _ys.size();
323
324 switch(vert)
325 {
326 case 0:
327 return base;
328 case 1:
329 return base + 1;
330 case 2:
331 return base + _xs.size();
332 case 3:
333 return base + _xs.size() + 1;
334 case 4:
335 return base + z_offset;
336 case 5:
337 return base + 1 + z_offset;
338 case 6:
339 return base + _xs.size() + z_offset;
340 case 7:
341 return base + _xs.size() + 1 + z_offset;
342 default:
343 std::abort();
344 }
345 }

354 std::uint64_t subdivision_level(std::size_t idx)
355 {
356 Point vtx = vertex(idx);
357
358 std::uint64_t level = 0;
359
360 for(std::size_t slice(0); slice < _zs.size() - 1; slice++)
361 {
362 if(_zs[slice].coord <= vtx.z && vtx.z <= _zs[slice + 1].coord)
363 {
364 level = std::max(_zs[slice].subdivision_level, level);
365 }
366 }
367
368 return level;

379 void write_feat_xml(std::ostream& stream) const
380 {
381 stream << "<FeatMeshFile version=\"1\" mesh=\"conformal:hypercube:3:3\">\n";
382 stream << "<Mesh type=\"conformal:hypercube:3:3\" size=\""
383 << num_vertices() << " "
384 << num_edges() << " "
385 << num_faces() << " "
386 << num_cells() << "\">\n";
387 stream << "<Vertices>\n";

389 {

391 stream << point.x << " " << point.y << " " << point.z << "\n";
392 }
393 stream << "</Vertices>\n";
394
395 stream << "<Topology dim=\"1\">\n";
396 for(std::size_t i(0); i < num_edges(); i++)
397 {
398 for(std::size_t j(0); j < 2; j++)
399 {
400 stream << edge(i, j);
401 stream << (j == 1 ? "\n" : " ");
402 }
403 }
404 stream << "</Topology>\n";
405
406 stream << "<Topology dim=\"2\">\n";
407 for(std::size_t i(0); i < num_faces(); i++)
408 {
409 for(std::size_t j(0); j < 4; j++)
410 {
411 stream << face(i, j);
412 stream << (j == 3 ? "\n" : " ");
413 }
414 }
415 stream << "</Topology>\n";
416
417 stream << "<Topology dim=\"3\">\n";
418 for(std::size_t i(0); i < num_cells(); i++)
419 {
420 for(std::size_t j(0); j < 8; j++)
421 {
422 stream << cell(i, j);
423 stream << (j == 7 ? "\n" : " ");
424 }
425 }
426 stream << "</Topology>\n";

428 stream << "</Mesh>\n";
429 stream << "</FeatMeshFile>\n";
