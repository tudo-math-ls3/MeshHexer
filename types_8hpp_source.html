211 std::size_t edge(std::size_t i, std::size_t j) const
212 {
213 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
214 // FEATs structured mesh counts fences, rather than fence posts.
215 // Hence the +- 1 everywhere
216
217 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
218
219 // auxiliary index
220 std::size_t pos = 0;
221
222 // z-direction
223 if(
224 i >= num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) +
225 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
226 {
227 pos = i - num_slices[1] * (num_slices[0] + 1) * (num_slices[2] + 1) -
228 (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
229 return (pos / num_slices[2]) + ((pos % num_slices[2] + j) * (num_slices[0] + 1) * (num_slices[1] + 1));
230 }
231 // y-direction
232 else if(i >= (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0])
233 {
234 pos = i - (num_slices[2] + 1) * (num_slices[1] + 1) * num_slices[0];
235 std::size_t x = (pos / num_slices[1]) % (num_slices[0] + 1);
236 std::size_t y = pos % num_slices[1];
237 std::size_t z = (pos / num_slices[1]) / (num_slices[0] + 1);
238 return (z * (num_slices[0] + 1) * (num_slices[1] + 1)) + ((y + j) * (num_slices[0] + 1)) + x;
239 }
240 // x-direction
241 else
242 {
243 return i + j + (i / num_slices[0]);
244 }
245 }

255 std::size_t face(std::size_t i, std::size_t j) const
256 {
257 // NOTE(mmuegge): Copied from FEAT3 (kernel/geometry/intern/struct_index_mapping.hpp)
258 // FEATs structured mesh counts fences, rather than fence posts.
259 // Hence the +- 1 everywhere
260
261 const std::array<std::size_t, 3> num_slices = {_xs.size() - 1, _ys.size() - 1, _zs.size() - 1};
262
263 std::size_t pos = 0;
264 if(i < num_slices[0] * num_slices[1] * (num_slices[2] + 1))
265 {
266 return i + (i / num_slices[0]) + ((i / (num_slices[0] * num_slices[1])) * (num_slices[0] + 1)) + (j % 2) +
267 ((j / 2) * (num_slices[0] + 1));
268 }
269 else if(
270 i >= num_slices[0] * num_slices[1] * (num_slices[2] + 1) &&
271 i < num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1)))
272 {
273 pos = i - num_slices[0] * num_slices[1] * (num_slices[2] + 1);
274 return (pos % num_slices[0]) + ((pos / (num_slices[0] * num_slices[2])) * (num_slices[0] + 1)) +
275 ((pos % (num_slices[0] * num_slices[2])) / num_slices[0] * (num_slices[0] + 1) * (num_slices[1] + 1)) +
276 (j % 2) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
277 }
278 else
279 {
280 pos = i - num_slices[0] * (num_slices[1] * (num_slices[2] + 1) + num_slices[2] * (num_slices[1] + 1));
281 return (pos / (num_slices[1] * num_slices[2])) + ((pos % num_slices[1]) * (num_slices[0] + 1)) +
282 (((pos / num_slices[1]) % num_slices[2]) * (num_slices[0] + 1) * (num_slices[1] + 1)) +
283 ((j % 2) * (num_slices[0] + 1)) + ((j / 2) * (num_slices[0] + 1) * (num_slices[1] + 1));
284 }
285 }

295 std::size_t cell(std::size_t idx, std::size_t vert) const
296 {
297 const std::size_t z_layer = idx / ((_xs.size() - 1) * (_ys.size() - 1));
298 const std::size_t y_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) / (_xs.size() - 1);
299 const std::size_t x_layer = idx % ((_xs.size() - 1) * (_ys.size() - 1)) % (_xs.size() - 1);
300
301 const std::size_t base = (z_layer * (_xs.size() * _ys.size())) + (y_layer * _xs.size()) + x_layer;
302 const std::size_t z_offset = _xs.size() * _ys.size();
303
304 switch(vert)
305 {
306 case 0:
307 return base;
308 case 1:
309 return base + 1;
310 case 2:
311 return base + _xs.size();
312 case 3:
313 return base + _xs.size() + 1;
314 case 4:
315 return base + z_offset;
316 case 5:
317 return base + 1 + z_offset;
318 case 6:
319 return base + _xs.size() + z_offset;
320 case 7:
321 return base + _xs.size() + 1 + z_offset;
322 default:
323 std::abort();
324 }
325 }

334 std::uint64_t subdivision_level(std::size_t idx)
335 {
336 Point vtx = vertex(idx);
337
338 std::uint64_t level = 0;
339
340 for(std::size_t slice(0); slice < _zs.size() - 1; slice++)
341 {
342 if(_zs[slice].coord <= vtx.z && vtx.z <= _zs[slice + 1].coord)
343 {
344 level = std::max(_zs[slice].subdivision_level, level);
345 }
346 }
347
348 return level;

359 void write_feat_xml(std::ostream& stream) const

363 << num_vertices() << " "

371 stream << point.x << " " << point.y << " " << point.z << "\n";
372 }
373 stream << "</Vertices>\n";
374
375 stream << "<Topology dim=\"1\">\n";
376 for(std::size_t i(0); i < num_edges(); i++)
377 {
378 for(std::size_t j(0); j < 2; j++)
379 {
380 stream << edge(i, j);
381 stream << (j == 1 ? "\n" : " ");
382 }
383 }
384 stream << "</Topology>\n";
385
386 stream << "<Topology dim=\"2\">\n";
387 for(std::size_t i(0); i < num_faces(); i++)
388 {
389 for(std::size_t j(0); j < 4; j++)
390 {
391 stream << face(i, j);
392 stream << (j == 3 ? "\n" : " ");
393 }
394 }
395 stream << "</Topology>\n";
396
397 stream << "<Topology dim=\"3\">\n";
398 for(std::size_t i(0); i < num_cells(); i++)
399 {
400 for(std::size_t j(0); j < 8; j++)
401 {
402 stream << cell(i, j);
403 stream << (j == 7 ? "\n" : " ");
404 }
405 }
406 stream << "</Topology>\n";
407
408 stream << "</Mesh>\n";
409 stream << "</FeatMeshFile>\n";
